void ms_delay(unsigned int time)
{
	unsigned int i,j;
	for(i=0;i<time;i++)
	for(j=0;j<1275;j++);
}

unsigned char en_ip_fun(bit ip1,bit ip2)
{
	unsigned char temp;
	if(ip1==0 & ip2==0)
	{
		en_ip1=0;
		ms_delay(1);
		temp=P0;
		en_ip1=1;
	}
	if(ip1==0 & ip2==1)
	{
		en_ip2=0;
		ms_delay(1);
		temp=P0;
		en_ip2=1;
	}
	if(ip1==1 & ip2==0)
	{
		en_ip3=0;
		ms_delay(1);
		temp=P0;
		en_ip3=1;
	}
	if(ip1==1 & ip2==1)
	{
		en_ip4=0;
		ms_delay(1);
		temp=P0;
		en_ip4=1;
	}
	return temp;
}

void en_op_fun(bit op,unsigned char temp)
{
	if(op==0)
	{
	  P0=temp;
	  en_op1=1;
	  ms_delay(1);
	  en_op1=0;
	}
	else
	{
	  P0=temp;
	  en_op2=1;
	  ms_delay(1);
	  en_op2=0;	
	}
}

{

	temp2= ip4 & 0x04;

	else
	{
	  if(disp==0)
 	  {
	    switch(car)

		    case 0x01:

		    case 0x02:

		    case 0x04:
		      temp = 0xA4;
		      break;
		    case 0x08:

		      break;
		    case 0x10:
		      temp = 0x99;

		      break;
		    case 0x40:
